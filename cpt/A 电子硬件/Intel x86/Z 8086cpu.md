//只能实现16位寄存器的mov add
#include <stdio.h>
#define REGISTER unsigned short

REGISTER ax=0,bx=0,cx=0,dx=0, si=0,di=0,sp=0,bp=0, cs=0,ss=0,ds=0,es=0;
REGISTER ip;

void mov(REGISTER &reg1,REGISTER &reg2){
    reg1 = reg2;
}

void add(REGISTER &reg1,REGISTER &reg2){
    reg1 = reg1 + reg2;
}

void show(){
    printf("ax\t%XH\t",ax);
    printf("bx\t%XH",bx);
    printf("\n");
}

int main(){

    REGISTER tp;
    tp=0x4E20;
    mov(ax,tp);
    show();
    tp=0x1406;
    add(ax,tp);
    show();
    tp=0x2000;
    mov(bx,tp);
    show();
    add(ax,bx);
    show();
    mov(bx,ax);
    show();
    add(ax,bx);

    show();

    return 0;
}
