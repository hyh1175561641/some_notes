# 系统异常处理


```java

//自定义异常信息
public class QyException extends RuntimeException{
  private String code;//状态码
  private String msg;//状态提示信息
  public QyException(String code, String msg) {
    this.code = code;
    this.msg = msg;
  }
}

//异常拦截器
@ControllerAdvice
public class SysExceptionHandler {
    //处理自定义异常的方法
    @ExceptionHandler(QyException.class)
    @ResponseBody
    public Result processQyException(Throwable cause, HttpServletRequest request, HttpServletResponse response){
        QyException qyException=null;
        if(EmptyUtils.isNotEmpty(cause) && cause instanceof  QyException){
            qyException= (QyException) cause;
            //返回自定义异常信息
            return  ResultUtils.returnFail(qyException.getMsg(), qyException.getCode());
        }else{
            return ResultUtils.returnFail();
        }
    }
    //处理系统异常的方法
    @ExceptionHandler(Exception.class)
    public void processCommonException(Throwable cause, HttpServletRequest request, HttpServletResponse response){
        cause.printStackTrace();
        try {
            //返回500页面
            response.sendRedirect(request.getContextPath()+"/500.jsp");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}












```
